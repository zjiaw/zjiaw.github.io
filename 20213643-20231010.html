var videos = [ "rem.mp4","zerptwo.mp4","selena.mp4", "saber.mp4" ]; var videoIndex = 0; var videoElement = document.getElementById('myVideo'); var playPauseButton = document.getElementById('play-pause-button'); videoElement.src = videos[videoIndex]; function previous() { if (videoIndex > 0) { videoIndex--; videoElement.src = videos[videoIndex]; } } function next() { if (videoIndex < videos.length - 1) { videoIndex++; videoElement.src = videos[videoIndex]; } } function togglePlayPause() { if (videoElement.paused) { videoElement.play(); playPauseButton.innerHTML = '暂停'; } else { videoElement.pause(); playPauseButton.innerHTML = '播放'; } } function playRandomVideo() { videoElement.pause(); videoIndex = Math.floor(Math.random() * videos.length); videoElement.src = videos[videoIndex]; videoElement.play(); } playPauseButton.addEventListener('click', togglePlayPause);
